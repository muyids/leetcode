#### [剑指 Offer II 085. 生成匹配的括号](https://leetcode.cn/problems/IDBivT/)



#### DFS

记录左括号和右括号的个数

```cpp
class Solution {
public:
    vector<string> ans;

    vector<string> generateParenthesis(int n) {
        dfs(0, 0, 0, n, "");
        return ans;
    }

    void dfs(int l, int r, int i, int n, string path){
        if (i == n * 2){
            ans.push_back(path);
            return;
        }
        if (l < n)  dfs(l+1, r, i+1, n, path+"(");
        if (r < l)  dfs(l, r+1, i+1, n, path+")");
    }
};
```
