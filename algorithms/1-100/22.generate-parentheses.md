给出 n 代表生成括号的对数，请你写出一个函数，使其能够生成所有可能的并且有效的括号组合。

```cpp
例如，给出 n = 3，生成结果为：

[
  "((()))",
  "(()())",
  "(())()",
  "()(())",
  "()()()"
]
```

## DFS

记录左括号和右括号的个数

```cpp
class Solution {
public:
    vector<string> ans;

    vector<string> generateParenthesis(int n) {
        dfs(0, 0, 0, n, "");
        return ans;
    }

    void dfs(int l, int r, int i, int n, string path){
        if (i == n * 2){
            ans.push_back(path);
            return;
        }
        if (l < n)  dfs(l+1, r, i+1, n, path+"(");
        if (r < l)  dfs(l, r+1, i+1, n, path+")");
    }
};
```
