
### 解题思路

dfs

### 代码

```cpp
class Solution {
public:
    vector<int> path;
    vector<bool> used;
    vector<vector<int>> ans;
    int N;
    vector<vector<int>> permute(vector<int>& nums) {
        N = nums.size();
        used = vector<bool>(N, false);
        dfs(0, nums);
        return ans;
    }

    void dfs(int d, vector<int>& nums){
        if (d == nums.size()) {
            ans.push_back(path);
            return ;
        }
        for (int i = 0; i < nums.size(); i++){ // 把第i个位置的数字依次加到结果集后面
            if (!used[i]){
                used[i] = true;
                path.push_back(nums[i]);
                dfs(d+1, nums);
                path.pop_back();
                used[i] = false;
            }
        }
    }
};
```
