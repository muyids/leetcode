/**
 * @param {number} n
 * @param {number} k
 * @return {string}
 */
var getPermutation = function (n, k) {

    let A = []
    for (let i = 0; i < n; i++) {
        A[i] = i + 1
    }


    function minIdx(i) {
        let j = A[i-1]
        let min = A[i], idx = i
        for (; i < A.length; i++) {
            if (A[i] < min && A[i] > j) {
                min = A[i]
                idx = i
            }
        }
        return idx
    }

    function reverse(i) {
        let start = i, end = A.length - 1
        while (start < end) {
            swap(start++, end--)
        }

    }

    function swap(i, j) {
        let t = A[i]
        A[i] = A[j]
        A[j] = t
    }

    function nextPermutation(nums) {
        for (let i = nums.length - 1; i > 0; i--) {
            if (nums[i] > nums[i - 1]) {
                swap(i - 1, minIdx(i))
                reverse(i)
                break
            }
        }
        return nums
    }
    while (--k > 0) {
        A = nextPermutation(A)
    }

    return A.join('')
};
