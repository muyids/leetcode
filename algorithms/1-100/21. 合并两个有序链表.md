将两个有序链表合并为一个新的有序链表并返回。新链表是通过拼接给定的两个链表的所有节点组成的。 

```case
示例：

输入：1->2->4, 1->3->4
输出：1->1->2->3->4->4
```

---

递归遍历

```javascript
var mergeTwoLists = function(l1, l2) {
    if (!l1) return l2
    if (!l2) return l1
    let head = null
    if (l2.val< l1.val) {
        head = l2
        l2 = l2.next
    }else {
        head = l1
        l1 = l1.next
    }
    let cur = head
    while (l1 && l2){
        if (l1.val > l2.val){
            cur.next = l2
            l2 = l2.next
        } else {
            cur.next = l1
            l1 = l1.next
        }
        cur = cur.next
    }
    if (!l1) cur.next = l2
    else cur.next = l1
    return head
};
```
