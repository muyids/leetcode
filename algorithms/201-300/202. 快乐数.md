/**
 * @param {number} n
 * @return {boolean}
 */
var isHappy = function (n) {

    let map = {}
    while (n !== 1) {
        n = n.toString().split('').reduce((pre, i) => {
            return pre + i * i
        }, 0)
        if (map[n]){
            return false
        }
        map[n] = true
    }

    return true
};

let r = isHappy(2)
