给定一个字符串 s ，找出 至多 包含两个不同字符的最长子串 t 。

```cpp
示例 1:

输入: "eceba"
输出: 3
解释: t 是 "ece"，长度为3。
示例 2:

输入: "ccaabbb"
输出: 5
解释: t 是 "aabbb"，长度为5。
```

---

滑动窗口

cpp

```cpp
class Solution {
public:
    int lengthOfLongestSubstringTwoDistinct(string s) {
        int l = 0;
        int r = 0;
        unordered_map<char, int> window;
        int maxLen =0;
        while(r < s.size()){
            window[s[r]]++;
            r++;
            while (window.size() > 2){
                window[s[l]]--;
                if (window[s[l]] == 0){
                    window.erase(s[l]);
                }
                l++;
            }
            maxLen = max(maxLen, r - l);
        }
        return maxLen;
    }
};
```
