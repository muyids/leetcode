
把中序遍历的非递归写法拆开

```javascript
var BSTIterator = function(root) {
    this.stack = []
    while (root){
        this.stack.push(root)
        root = root.left
    }
};

BSTIterator.prototype.next = function() {
    let p = this.stack.pop()
    let res = p.val
    p = p.right
    while(p){
        this.stack.push(p)
        p = p.left
    }
    return res
};

BSTIterator.prototype.hasNext = function() {
    return this.stack.length > 0
};
```
