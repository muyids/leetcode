给定一个非空字符串 s 和一个包含非空单词列表的字典 wordDict，在字符串中增加空格来构建一个句子，使得句子中所有的单词都在词典中。返回所有这些可能的句子。

说明：

- 分隔时可以重复使用字典中的单词。
- 你可以假设字典中没有重复的单词。

```cpp
示例 1：

输入:
s = "catsanddog"
wordDict = ["cat", "cats", "and", "sand", "dog"]
输出:
[
  "cats and dog",
  "cat sand dog"
]
示例 2：

输入:
s = "pineapplepenapple"
wordDict = ["apple", "pen", "applepen", "pine", "pineapple"]
输出:
[
  "pine apple pen apple",
  "pineapple pen apple",
  "pine applepen apple"
]
解释: 注意你可以重复使用字典中的单词。
示例 3：

输入:
s = "catsandog"
wordDict = ["cats", "dog", "sand", "and", "cat"]
输出:
[]
```

看到题目，我们首先想到的是dfs记忆化搜索，遍历s[i:j]的所有子串，判断在字典中是否出现，如果出现，记录区间[i, j], 最后dfs搜索所有路径，

时间复杂度为 O(N^2*M), N为s的长度，M为字典的大小

dp[i][j]来保存s[i:j]是否在字典里出现

用字典树优化判断字符串出现的效率


```javascript
/**
 * @param {string} s
 * @param {string[]} wordDict
 * @return {string[]}
 */
var wordBreak = function (s, wordDict) {
  let dp = new Array(s.length)
  for (let i = 0; i < dp.length; i++) {
    dp[i] = new Array(s.length).fill(0)
  }

  let trie = new Trie()
  for (let i = 0; i < wordDict.length; i++) {
    trie.insert(wordDict[i])
  }

  // dp
  for (let i = 0; i < s.length; i++) {
    let cur = trie.root
    for (let j = i; j < s.length; j++) {
      if (cur.next[s[j]]) {
        cur = cur.next[s[j]]
        if (cur.end) {
          dp[i][j] = 1
        }
      } else {
        break
      }
    }
  }

  // 深搜
  let ans = []

  function dfs(i, path) {
    if (i == s.length) {
      ans.push(path.join(' '))
      return
    }
    for (let j = i; j < s.length; j++) {
      if (dp[i][j]) {
        dfs(j + 1, [...path, s.substr(i, j - i + 1)])
      }
    }
  }
  dfs(0, [])
  return ans
};
```

上面算法 有一种case 会超时

```cpp
s = "aaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaa"
wordDict = ["a", "aa", "aaa", "aaaa", "aaaaa", "aaaaaa", "aaaaaaa", "aaaaaaaa", "aaaaaaaaa", "aaaaaaaaaa"]
```

这种情况，dp数组里保存了[i:j]的所有可能情况，让深搜查找了大量无效路径，实际上[i:j]存在于字典集，也不一定出现在句子里

自顶向下 的算法是不可取的，我们反其道而行，思考自底向上可不可以

我们用`dp[j] = [i1, i2, i3....]`保存可以到达位置j的所有开始位置`i1, i2, i3....`，这样可以大大减少需要深搜的路径数


```javascript
/**
 * @param {string} s
 * @param {string[]} wordDict
 * @return {string[]}
 */
var wordBreak = function (s, wordDict) {

    let trie = new Trie()
    for (let i = 0; i < wordDict.length; i++) {
        trie.insert(wordDict[i])
    }

    // 构建动规状态
    let dp = new Array(s.length + 1) // dp[j] 表示[i,j)的起始位置集合
    dp[0] = [-1]
    for (let i = 0; i < s.length; i++) {
        if (!dp[i]) continue // 到达不了当前位置，剪枝
        let cur = trie.root
        for (let j = i; j < s.length; j++) {
            if (cur.next[s[j]]) {
                cur = cur.next[s[j]]
                if (cur.end) {
                    if (dp[j + 1]) dp[j + 1].push(i)
                    else dp[j + 1] = [i]
                }
            } else {
                break
            }
        }
    }
    // 深搜
    let ans = []

    function dfs(j, path) {
        if (j == 0) {
            ans.push(path.join(' '))
            return
        }
        if (!dp[j]) return
        for (let i of dp[j]) {
            dfs(i, [s.substr(i, j - i), ...path])
        }
    }
    dfs(s.length, [])
    return ans
};

// 构建字典树
const Trie = function () {
    this.root = new Node('')
}

const Node = function (val) {
    this.val = val
    this.next = {}
    this.end = false
}

Trie.prototype.insert = function (word) {
    let cur = this.root
    for (let i = 0; i < word.length; i++) {
        if (!cur.next[word[i]]) {
            cur.next[word[i]] = new Node(word[i])
        }
        cur = cur.next[word[i]]
        if (i == word.length - 1) cur.end = true
    }
}
```

cpp代码

```cpp

class Solution {
public:
    struct Node {
        map<char, Node *> next;
        bool end = false;
        Node() : end(false) {};
    };

    Node *root;

    void insert(string word) {
        auto cur = root;
        for (int i = 0; i < word.size(); i++) {
            if (cur->next.count(word[i]) == 0) {
                cur->next[word[i]] = new Node;
            }
            cur = cur->next[word[i]];
            if (i == word.size() - 1)  cur->end = true;
        };
    }

    void dfs(string s, vector<vector<int >> &dp, int i, vector<string> &path, vector<string> &res) {
        if (i == 0) {
            string t;
            for (auto it = path.rbegin(); it != path.rend(); ++it) {
                t += *it + " ";
            }
            t.pop_back();
            res.push_back(t);
            return;
        }
        for (auto start: dp[i]) {
            path.push_back(s.substr(start, i - start));
            dfs(s, dp, start, path, res);
            path.pop_back();
        }
    }

    vector<string> wordBreak(string s, vector<string> &wordDict) {
        // 构建字典树
        vector<string> res;
        root = new Node;
        for (int i = 0; i < wordDict.size(); i++) {
            insert(wordDict[i]);
        }

        unordered_set<char> cs;
        for (auto &w : wordDict) {
            for (auto c : w) {
                cs.insert(c);
            }
        }
        for (auto c : s) {
            if (cs.count(c) == 0) {
                return vector<string>();
            }
        }

        // 动态规划
        vector<vector<int>> dp(s.size() + 1);
        dp[0].push_back(-1);
        for (int i = 0; i < s.size(); i++) {
            if (dp[i].size() == 0) {
                continue;
            }
            auto cur = root;
            for (int j = i; j < s.size(); j++) {
                if (cur->next.count(s[j]) > 0) {
                    cur = cur->next[s[j]];
                    if (cur->end) {
                        dp[j+1].push_back(i);
                    }
                } else {
                    break;
                }
            }
        }

        // 深度优先搜索
        vector<string> path;
        dfs(s, dp, s.size(), path, res);
        return res;
    };

};
```
