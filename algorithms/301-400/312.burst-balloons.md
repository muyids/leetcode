class Solution {
public:
    int maxCoins(vector<int>& nums) {
        nums.insert(nums.begin(),1);
        nums.push_back(1);
        int n=nums.size();
        int dp[n][n];   //dp[i][j]表示第i至第j个元素这个区间能获得的最大硬币数
        for(int i=0;i<n;i++)
            for(int j=0;j<n;j++)
                dp[i][j]=0;
        for(int r=2;r<n;r++)            //r为区间长度
            for(int i=0;i<n-r;i++){    //i为左区间
                int j=i+r;            //j为右区间
                for(int k=i+1;k<j;k++)
                    dp[i][j]=max(dp[i][j],dp[i][k]+dp[k][j]+nums[i]*nums[k]*nums[j]);
            }
        
        return dp[0][n-1];
    }
};
