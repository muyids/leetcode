
### 统计次数+顺序输出

统计T中个字母出现次数

遍历模板串，输出

### 代码

```cpp
class Solution {
public:
    string customSortString(string S, string T) {
        int A[26] = {0};
        for (char c : T) A[c-'a']++;

        string ans;
        for (char c: S){
            ans.append(A[c - 'a'], c);
            A[c - 'a'] = 0;
        }
        for (int i =0; i< 26; i++){
            if (A[i] > 0) ans.append(A[i], 'a' + i);
        }
        return ans;
    }
};
```
