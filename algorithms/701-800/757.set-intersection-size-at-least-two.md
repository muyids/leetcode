class Solution {
    public int intersectionSizeTwo(int[][] w) {

        Arrays.sort(w, new Comparator<int[]>(){
            @Override
            public int compare(int[] o1, int o2[]){
                if (o1[1] == o2[1]){
                    return o2[0] - o1[0];
                }
                return o1[1] - o2[1];
            }
        });

        List<Integer> s = new ArrayList<>();
        s.add(-1);
        int cnt = 0;
        for (int[] t: w){
            if (t[0] > s.get(cnt)) {
                s.add(t[1] -1);
                s.add(t[1]);
                cnt+=2;
            } else if (t[0] > s.get(cnt -1)){
                s.add(t[1]);
                cnt++;
            }
        }
        return cnt;
    }
}
