## cpp Set有序性

cpp set是按字典序排序的

1. 所有产品加入set
2. 从左到右枚举输入串
3. 迭代所有商品，删除不满足当前输入串的商品（不会满足更长的搜索条件，故删除）
4. 满足条件的前三个商品加入结果集

### 代码

```cpp
class Solution {
public:
    vector<vector<string>> suggestedProducts(vector<string> &products, string searchWord) {
        vector<vector<string>> ans;
        set<string> st;

        for (int i = 0; i < products.size(); i++) st.insert(products[i]);

        for (int i = 0; i < searchWord.length(); i++) {
            for (auto t = st.begin(); t != st.end(); t++) {
                if ((*t)[i] != searchWord[i]) st.erase(t);
            }
            int cnt = 0;
            vector<string> res;
            for (auto t = st.begin(); t != st.end() && cnt < 3; t++) {
                res.push_back(*t);
                cnt++;
            }
            ans.push_back(res);
        }
        return ans;
    }
};
```

struct treeNode {
    bool isEnd;
    treeNode *next[26];

    treeNode() {
        isEnd = false;
        memset(next, 0, sizeof(next));
    }
};

class Trie {

public:
    treeNode *root;

    Trie() {
        root = new treeNode();
    }

    void insert(const string &word) {
        treeNode *node = root;
        for (char c : word) {
            if (node->next[c - 'a'] == NULL) {
                node->next[c - 'a'] = new treeNode();
            }
            node = node->next[c - 'a'];
        }
        node->isEnd = true;
    }

    bool search(const string &word) {
        treeNode *node = root;
        for (char c : word) {
            node = node->next[c - 'a'];
            if (node == NULL) {
                return false;
            }
        }
        return node->isEnd;
    }

    bool startsWith(const string &prefix) {
        treeNode *node = root;
        for (char c : prefix) {
            node = node->next[c - 'a'];
            if (node == NULL) {
                return false;
            }
        }
        return node != NULL;
    }
};

class Solution {
public:
    vector<vector<string>> ans;
    vector<string> paths;
    vector<vector<string>> suggestedProducts(vector<string>& products, string searchWord) {
        Trie* t = new Trie();
        for (auto p : products){
            t->insert(p);
        }

        for (int i = 1; i<= searchWord.size(); i++){
            string s = searchWord.substr(0, i);
            dfs(t->root, "", s, 0);
            ans.push_back({paths.begin(), paths.end()});
            paths.clear();
        }
        return ans;
    }

    void dfs(treeNode* root, string rt, string s, int i){
        if (root == NULL) return;
        if (paths.size() == 3) return ;

        treeNode *node = root;
        if (!node) return;
        if (node->isEnd) paths.push_back(rt);
        if (i < s.size()){
            node = node->next[s[i] - 'a'];
            if (node == NULL) return;
            dfs(node, rt + s[i], s, i+1);
        } else {
            for (int j = 0; j< 26; j++){
                if (node->next[j]){
                    string z = rt;
                    z.append(1, 'a' + j );
                    dfs(node->next[j],z, s, i+1);
                }
            }
        }
    }
};
